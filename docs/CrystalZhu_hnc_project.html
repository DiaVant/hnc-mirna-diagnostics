Project Background

MiRNA expression profiling in tumor tissues provide invalvuable insight for cancer dianogiss (National Institute of Health). This project focuses on Head and Neck Cancer, historically hard to dianogse due to its vague symptoms and mimicking of common colds/illnesses. MiRNA data used was exported and cleaned from the National Cancer Institute Genomic Data Commons Data Portal.

T Tests

Mean, standard deviation, and the 2-tail student t-test p-value of Tumor and Control groups

meanV = rowMeans(dfM)

sdV = apply(dfM, 1, sd)

pV = c()
for (i in 1:dim(dfM)[1] ) {
  pV = c(pV, t.test( dfM[i,grep("Tumor",colnames(dfM))], dfM[i,grep("Control",colnames(dfM))], na.rm=T )$p.value  )
}

miTable = cbind.data.frame(meanV, sdV, pV)

row.names(miTable) = row.names(dfM)
colnames(miTable) = c("mean", "std dev", "p-value")

Find 50 most significantly different miRNA names between tumor and control samples

topPV = head(order(miTable$`p-value`), 50)
topTable = data.frame()

for (i in 1:length(topPV) ) {
  topTable = rbind(topTable, miTable[topPV[i],] )
}

Random Forest

Train random forest to classify tissue from measured miRNA levels

sig_table = data.frame()

for (i in 1:dim(miTable)[1] ) {
  p = miTable[i,"p-value"]
  if (!is.na(p) && p < 0.001) {
    sig_table = rbind(sig_table, df[i,] )
  }
}

model_sigTable = t(sig_table)

maxi = dim(model_sigTable)[1]

conditionV = row.names(model_sigTable)

for (i in 1:length(conditionV)) {
  if (grepl("Tumor", conditionV[i], fixed = T) ) {
    conditionV[i] = "Tumor"
  }
  else {
    conditionV[i] = "Control"
  }
}

Condition = as.factor(conditionV)
model_sigTable = cbind.data.frame(Condition, model_sigTable)
colnames(model_sigTable) = gsub("-", ".", colnames(model_sigTable) )

set.seed(4293)
trainInd = sample(1:maxi, round(.7 * maxi), replace=F)

trainXy = model_sigTable[trainInd,]

testXy = model_sigTable[-trainInd,]

HNSC_model = randomForest(Condition ~ . ,
                          ntree = 100,
                          data = trainXy
                          )
#importance(HNSC_model)[order( importance(HNSC_model)), ]
varImpPlot(HNSC_model)
